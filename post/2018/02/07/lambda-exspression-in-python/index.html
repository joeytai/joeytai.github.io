 import time
start = time.clock()
def fib(n):
    if n <= 2: return 1
    else:
        return fib(n-1) + fib(n-2)
print (fib(20))
end = time.clock()
print ("read: %f s" % (end - start))

6765
read: 0.004619 s

start = time.clock()
def fib(n):
    return n <= 2 and 1 or fib(n - 1) + fib(n - 2)
print (fib(20))
end = time.clock()
print ("read: %f s" % (end - start))

6765
read: 0.002924 s
